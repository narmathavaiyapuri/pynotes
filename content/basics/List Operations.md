```python
# Add and remove elements
numbers = [1, 2, 3]

numbers.remove(2)

```

```python
# List slicing
nums = [10, 20, 30, 40, 50]

print(nums[1:4])
print(nums[:3])
```

    [20, 30, 40]
    [10, 20, 30]
    


```python
# Loop and sum
marks = [80, 90, 70]

total = 0

for m in marks:
    total += m

print("Total:", total)
```

    Total: 240
