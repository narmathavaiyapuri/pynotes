Dictionaries

Tue 21 April 2026 
# Add and update values
user = {"name": "Narmatha", "age": 21}

user["age"] = 22
user["city"] = "Chennai"

print(user)

{'name': 'Narmatha', 'age': 22, 'city': 'Chennai'}

# Remove element
data = {"a": 10, "b": 20, "c": 30}

data.pop("b")

print(data)

{'a': 10, 'c': 30}

# Loop through dictionary
marks = {"math": 80, "science": 90}

for subject, mark in marks.items():
    print(subject, mark)

math 80
science 90
