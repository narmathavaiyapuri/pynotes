List Operations

Mon 20 April 2026 
# Add and remove elements
numbers = [1, 2, 3]

numbers.append(4)
numbers.remove(2)

print(numbers)

[1, 3, 4]

# List slicing
nums = [10, 20, 30, 40, 50]

print(nums[1:4])
print(nums[:3])

[20, 30, 40]
[10, 20, 30]

# Loop and sum
marks = [80, 90, 70]

total = 0

for m in marks:
    total += m

print("Total:", total)
